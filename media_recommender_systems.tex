\section{The Media Recommendation Problem}

\begin{frame}{Why Media Recommendation Is Fundamentally Different}
\begin{columns}[T]
\column{0.48\textwidth}
\small
\textbf{Job / E-commerce RS}
\begin{itemize}
    \item High-stakes: wrong job $\to$ years lost
    \item Sparse interactions (few apps/user)
    \item Non-stationary preferences (months)
    \item Multi-stakeholder optimization
    \item Long feedback cycles
\end{itemize}

\column{0.48\textwidth}
\small
\textbf{Media RS (Spotify, YouTube, Netflix)}
\begin{itemize}
    \item Low-stakes: wrong song $\to$ 30s skip
    \item Abundant implicit feedback (skips, dwell)
    \item Rapidly shifting tastes (minutes/hours)
    \item Single-stakeholder: user satisfaction
    \item Real-time feedback loops
\end{itemize}
\end{columns}

\vfill
\centering
\small
\textbf{Key insight}: Media RS must optimize for \textbf{engagement and session depth}, with instant feedback on every recommendation.
\end{frame}

%=================================================================
\begin{frame}{Spotify: Collaborative Filtering at Scale}
\small
\textbf{Matrix Factorization} for implicit feedback:
\begin{equation*}
\hat{r}_{u,i} = \mu + b_u + b_i + \mathbf{p}_u \cdot \mathbf{q}_i^T
\end{equation*}

\textbf{Weighted ALS} (implicit feedback):
\begin{equation*}
\min_{\mathbf{P},\mathbf{Q}} \sum_{u,i} c_{ui}(p_{ui} - \mathbf{p}_u\mathbf{q}_i^T)^2 + \lambda(\|\mathbf{P}\|^2 + \|\mathbf{Q}\|^2)
\end{equation*}
where $c_{ui} = 1 + \alpha \cdot \text{play\_count}_{ui}$ and $p_{ui} = \mathbb{I}[\text{played}_{ui} > 0]$.

\vspace{0.15cm}
\textbf{Session-based CF} (next-track prediction):
\begin{equation*}
P(\text{track}_{t+1} \mid \text{track}_t) \propto \frac{\text{count}(\text{track}_t \to \text{track}_{t+1})}{\text{count}(\text{track}_t \to \cdot)}
\end{equation*}

\vspace{0.15cm}
\textbf{Discover Weekly} combines:
\begin{itemize}
    \item \textbf{CF}: Similar users' playlists
    \item \textbf{Content}: Audio feature similarity
    \item \textbf{NLP}: Artist/blog text embeddings
\end{itemize}
Each user receives a unique 30-track playlist refreshed weekly.
\end{frame}

%=================================================================
\begin{frame}{YouTube: Candidate Generation with Deep Neural Networks}
\small
\textbf{Architecture}: Two-tower neural network (query tower + video tower).

\begin{equation*}
\mathbf{e}_u = f_\theta(\text{watch\_history}, \text{search\_history}, \text{demographics}) \in \mathbb{R}^d
\end{equation*}
\begin{equation*}
\mathbf{e}_v = g_\phi(\text{video\_features}, \text{channel}, \text{tags}) \in \mathbb{R}^d
\end{equation*}

\textbf{Score}: $\displaystyle \text{score}(u,v) = \frac{\mathbf{e}_u \cdot \mathbf{e}_v}{\|\mathbf{e}_u\|\|\mathbf{e}_v\|}$

\vspace{0.15cm}
\textbf{Training}: Sampled softmax with millions of negative samples:
\begin{equation*}
\mathcal{L} = -\sum_{(u,v^+)} \log \frac{e^{\text{score}(u,v^+)}}{\sum_{v \in \mathcal{N}(u)} e^{\text{score}(u,v)}}
\end{equation*}

\vspace{0.15cm}
\textbf{Key design choices}:
\begin{itemize}
    \item Watch history: average of video embeddings (fixed window)
    \item Demographics: age, gender, locale embedded separately
    \item Example age: features decay with time (freshness bias)
    \item Hashing trick for sparse IDs (compressed feature space)
\end{itemize}
\end{frame}

%=================================================================
\begin{frame}{YouTube: Ranking with Wide \& Deep}
\small
\textbf{Wide \& Deep architecture} (Google, 2016):
\begin{equation*}
\hat{y} = \sigma(\underbrace{\mathbf{w}_{\text{wide}}^T \mathbf{x}_{\text{wide}}}_{\text{memorization}} \;+\; \underbrace{\mathbf{W}_L\,\text{ReLU}(\cdots\,\text{ReLU}(\mathbf{W}_1\mathbf{x}_{\text{deep}} + \mathbf{b}_1)\cdots) + b_L}_{\text{generalization}})
\end{equation*}

\vspace{0.15cm}
\textbf{Wide side} (linear): Cross-product feature transformations
\begin{equation*}
\phi_k(\mathbf{x}) = \prod_{i=1}^d x_i^{c_{ki}}, \quad c_{ki} \in \{0,1\}
\end{equation*}
Captures explicit co-occurrence patterns (e.g., user watched X AND video is from channel Y).

\vspace{0.15cm}
\textbf{Deep side} (DNN): 3 hidden layers (1024 $\to$ 512 $\to$ 256) with ReLU.

\vspace{0.15cm}
\textbf{Loss}: Weighted logistic regression with watch-time weighting:
\begin{equation*}
\mathcal{L} = -\sum_i w_i\bigl[y_i\log\hat{y}_i + (1-y_i)\log(1-\hat{y}_i)\bigr]
\end{equation*}
$w_i$ = observed watch time of the video. Optimizes for expected watch time, not just click.
\end{frame}

%=================================================================
\begin{frame}{YouTube: Expected Watch Time Optimization}
\small
\textbf{Key insight}: YouTube optimizes for \textbf{watch time}, not clicks. A clickbait video with 5s watch is worse than a relevant video with 10min watch.

\textbf{Training label}: $\displaystyle \text{label} = \frac{\text{watch\_time}}{\text{video\_duration}}$

\vspace{0.15cm}
\textbf{Weighted logistic regression}:
\begin{equation*}
\text{odds}(y=1 \mid \mathbf{x}) \approx \mathbb{E}[T] \quad\text{(expected watch time)}
\end{equation*}

\vspace{0.15cm}
\textbf{Feature engineering for watch time}:
\begin{table}
\centering
\begin{tabular}{ll}
\toprule
\textbf{Feature} & \textbf{Description} \\
\midrule
Watch history & Last 50 videos watched (embedding avg) \\
Language affinity & User's preferred languages \\
Freshness & How recent the video is \\
Previous impressions & How often user saw \& ignored this \\
Time since last watch & Session stickiness signal \\
Device \& time & Mobile vs desktop, hour of day \\
\bottomrule
\end{tabular}
\end{table}

\vspace{0.1cm}
\textbf{Result}: 3x improvement in watch time over previous system.
\end{frame}

\section{Netflix's Architecture}

\begin{frame}{Netflix: Personalized Ranking with Context}
\small
\textbf{Netflix's challenge}: One account, multiple users, multiple devices, global catalog.

\textbf{Three-tier personalization}:
\begin{enumerate}
    \item \textbf{Which titles to show}: Two-tower retrieval (user $\times$ context)
    \item \textbf{How to show them}: Artwork personalization (different posters per user)
    \item \textbf{Row selection}: Which genre rows appear and in what order
\end{enumerate}

\vspace{0.15cm}
\textbf{Ranking architecture}:
\begin{equation*}
P(\text{play} \mid u, i, \text{context}) = \sigma(\mathbf{w}^T \cdot [\phi_{\text{personal}}(u,i) \; \oplus \; \phi_{\text{context}}(t) \; \oplus \; \phi_{\text{popular}}(i)])
\end{equation*}

\vspace{0.15cm}
\textbf{Context signals}:
\begin{itemize}
    \item Time of day (morning: shorter content, evening: movies)
    \item Day of week (weekend: series bingeing)
    \item Device (mobile: shorter, TV: longer)
    \item Number of people watching (single vs group viewing)
\end{itemize}
\end{frame}

\begin{align*}
P(\text{row}_r \mid u) &\propto \text{engagement}_u(\text{genres in } r) \\
\text{RowScore}(r, u) &= \text{affinity}_u(r) + \text{diversity\_bonus}(u, r)

\section{Algorithmic Approaches}

\begin{frame}{Content-Based Recommendation for Media}
\small
\textbf{TF-IDF for text metadata}:
$\displaystyle \; \text{tf-idf}(t,d) = \frac{n_{t,d}}{\sum_{t'}n_{t',d}} \times \log\!\left(\frac{N}{\text{df}_t}\right)+1$

\textbf{User profile (centroid)}:
$\displaystyle \; \mathbf{p}_u = \frac{1}{|I_u^+|}\sum_{i\in I_u^+} \mathbf{v}_i^{\text{tfidf}}$

\vspace{0.15cm}
\textbf{Audio feature similarity} (Spotify):
$\displaystyle \; \text{sim}_{\text{audio}}(i,j) = \frac{\mathbf{a}_i \cdot \mathbf{a}_j}{\|\mathbf{a}_i\|\|\mathbf{a}_j\|}$

\vspace{0.15cm}
\textbf{Performance comparison}:
\begin{table}
\centering
\begin{tabular}{lcc}
\toprule
\textbf{Method} & \textbf{Recall@10} & \textbf{NDCG@10} \\
\midrule
TF-IDF + Cosine & 0.38 & 0.34 \\
Audio Features (CNN) & 0.55 & 0.48 \\
Word2Vec Avg (lyrics) & 0.47 & 0.42 \\
Multi-modal Fusion & \textbf{0.68} & \textbf{0.61} \\
\bottomrule
\end{tabular}
\end{table}
\end{frame}

%=================================================================
\begin{frame}{Collaborative Filtering for Media}
\small
\textbf{Matrix Factorization (implicit)}:
\begin{align*}
\hat{r}_{u,i} &= \mu + b_u + b_i + \mathbf{p}_u \cdot \mathbf{q}_i^T \\
\text{WALS: } \min &\sum_{u,i} w_{ui}(p_{ui} - \mathbf{p}_u\mathbf{q}_i^T)^2 + \lambda(\|\mathbf{P}\|^2 + \|\mathbf{Q}\|^2)
\end{align*}

\vspace{0.15cm}
\textbf{Item-based CF (for "Fans also bought")}:
\begin{equation*}
s_{i,j} = \frac{|U_i \cap U_j|}{|U_i \cup U_j|} \quad\text{(Jaccard on listeners/viewers)}
\end{equation*}

\vspace{0.15cm}
\textbf{Session-based CF (next-item prediction)}:
\begin{equation*}
P(i_{t+1} \mid i_t) = \text{softmax}(\mathbf{W} \cdot \text{avg}(\mathbf{e}_{i_1}, \ldots, \mathbf{e}_{i_t}))
\end{equation*}

\vspace{0.15cm}
\textbf{Sequential pattern mining}:
\begin{align*}
\text{Transition matrix: } & M_{i,j} = \frac{\text{count}(i \to j)}{\text{count}(i \to \cdot)} \\
\text{Session score: } & \text{score}(i_{t+1}) = \sum_{k=1}^t \alpha^{t-k} M_{i_k, i_{t+1}}
\end{align*}
$\alpha$ = temporal decay (recent items weighted more).
\end{frame}

%=================================================================
\begin{frame}{Knowledge-Based Constraints in Media}
\small
\textbf{Hard constraints}:
\begin{align*}
\text{explicit\_block}(u, i) &\Rightarrow \text{exclude } \forall \text{ artists/genre} \\
\text{maturity\_rating}(i) &\leq \text{max\_rating}(u) \\
\text{geo\_availability}(i) &= \text{true} \quad\text{(licensing)}
\end{align*}

\vspace{0.15cm}
\textbf{Utility-based scoring}:
\begin{equation*}
U(u,i) = w_1 u_{\text{genre}} + w_2 u_{\text{mood}} + w_3 u_{\text{duration}} + w_4 u_{\text{popularity}}
\end{equation*}

\vspace{0.15cm}
\textbf{Playlist continuation (Spotify)}:
\begin{equation*}
\text{score}(\text{track}_t \mid \text{playlist}) = \lambda_{\text{CF}} \cos(\mathbf{e}_t, \bar{\mathbf{e}}_{\text{playlist}}) + \lambda_{\text{CB}} \text{sim}_{\text{audio}}(t, \text{playlist\_avg})
\end{equation*}

\vspace{0.15cm}
\textbf{Hybrid score (production)}:
\begin{equation*}
\text{Final} = \alpha_{\text{CB}}\,\text{score}_{\text{CB}} + \beta_{\text{CF}}\,\text{score}_{\text{CF}} + \gamma_{\text{KB}}\,\text{score}_{\text{KB}}
\end{equation*}
Weights tuned via Bayesian optimization per user segment.
\end{frame}

\section{The Skip \& Engagement Problem}

\begin{frame}{Why Users Skip — The Core Challenge}
\small
\textbf{Skip behavior}: The primary negative signal in media RS.

\begin{columns}[T]
\column{0.48\textwidth}
\textbf{Skip patterns by platform}:
\begin{itemize}
    \item \textbf{Spotify}: $\sim$30\% of tracks skipped within 10s
    \item \textbf{YouTube}: $\sim$40\% of videos abandoned within 30s
    \item \textbf{Netflix}: $\sim$20\% of titles abandoned within 5min
\end{itemize}

\column{0.48\textwidth}
\textbf{Why users skip}:
\begin{itemize}
    \item Wrong genre/mood
    \item Already heard/seen
    \item Too long/short
    \item Poor recommendation quality
    \item Context mismatch (time, device, company)
\end{itemize}
\end{columns}

\vspace{0.15cm}
\textbf{Modeling skips as feedback}:
\begin{equation*}
\text{Satisfaction}_{ui} = \begin{cases}
1 & \text{if play } > 70\% \text{ duration} \\
0 & \text{if skip } < 30\% \text{ duration} \\
\text{partial} & \text{otherwise}
\end{cases}
\end{equation*}

\textbf{Key insight}: A skip is not just "no" --- it's "wrong context." Same user may not skip the same song at a different time.
\end{frame}

%=================================================================
\begin{frame}{Modeling the Skip Signal}
\small
\textbf{Skip prediction as auxiliary task}:
\begin{equation*}
P(\text{skip} \mid u, i, \text{ctx}) = \sigma(\mathbf{w}^T \cdot \phi_{\text{skip}}(u,i,\text{ctx}))
\end{equation*}

\vspace{0.15cm}
\textbf{Multi-task learning architecture}:
\begin{equation*}
\mathcal{L} = \mathcal{L}_{\text{play}} + \lambda_{\text{skip}} \mathcal{L}_{\text{skip}} + \lambda_{\text{dwell}} \mathcal{L}_{\text{dwell}}
\end{equation*}

\vspace{0.15cm}
\textbf{Skip-aware ranking}:
\begin{equation*}
\text{RankScore}(u,i) = \text{Score}_{\text{base}}(u,i) - \beta \cdot P(\text{skip} \mid u,i)
\end{equation*}
$\beta$ controls how aggressively to demote likely-to-be-skipped items.

\vspace{0.15cm}
\textbf{Contextual skip modeling}:
\begin{table}
\centering
\begin{tabular}{ll}
\toprule
\textbf{Context} & \textbf{Skip Probability Change} \\
\midrule
Morning commute & +15\% (distracted listening) \\
Late night & -10\% (focused) \\
Party / social & +25\% (background) \\
Gym / workout & -20\% (high energy needed) \\
\bottomrule
\end{tabular}
\end{table}
\end{frame}

%=================================================================
\begin{frame}{Dwell Time as a Quality Signal}
\small
\textbf{Dwell time} = how long the user engaged before skipping. This is the most granular feedback signal.

\textbf{Weighted by dwell}:
\begin{equation*}
\text{weight}_{ui} = \frac{\min(\text{dwell}_{ui}, \text{cap})}{\text{duration}_i} \in [0, 1]
\end{equation*}

\vspace{0.15cm}
\textbf{Censored regression model} (YouTube):
\begin{equation*}
P(\text{watch\_time} > t \mid u,i) = \exp\!\left(-\frac{t}{\mu(u,i)}\right)
\end{equation*}
where $\mu(u,i) = \exp(\mathbf{w}^T \mathbf{x})$ is the expected watch time.

\vspace{0.15cm}
\textbf{Skip timing patterns}:
\begin{center}
\begin{tikzpicture}[scale=0.6]
\draw[->, thick] (0,0) -- (6,0) node[right] {Time};
\draw[->, thick] (0,0) -- (0,4) node[above] {Skip density};
\draw[blue, thick] plot[smooth] coordinates {(0,0.5) (1,2.5) (2,1.5) (3,1) (4,0.5) (5,0.2)};
\draw[dashed, red] (0.5,0) -- (0.5,3);
\node[below] at (0.5,-0.2) {10s};
\node at (4.5, 3) {Late skip = mismatch};
\node at (1, 3.2) {Early skip = disinterest};
\end{tikzpicture}
\end{center}

Early skips ($<10s$): user didn't like the recommendation at all.
Late skips ($>60\%$): user liked it initially but lost interest.
\end{frame}

\section{Session-Based \& Sequential Recommendation}

\begin{frame}{Session-Based Recommendation}
\small
\textbf{The session problem}: Users don't have persistent identities (especially YouTube). Only the current session matters.

\textbf{GRU4Rec} (Session-based RNN):
\begin{equation*}
\mathbf{h}_t = \text{GRU}(\mathbf{h}_{t-1}, \mathbf{e}_{i_t}), \quad P(i_{t+1}) = \text{softmax}(\mathbf{W} \cdot \mathbf{h}_t)
\end{equation*}

\vspace{0.15cm}
\textbf{Session features}:
\begin{align*}
\text{Session length:} &\quad \text{num\_clips}, \text{total\_time} \\
\text{Session diversity:} &\quad \text{unique\_genres} / \text{total\_clips} \\
\text{Session trajectory:} &\quad \text{genre transitions} \\
\text{Abandonment:} &\quad \text{last\_action}, \text{time\_since\_last}
\end{align*}

\vspace{0.15cm}
\textbf{Next-item prediction accuracy}:
\begin{table}
\centering
\begin{tabular}{lcc}
\toprule
\textbf{Method} & \textbf{Recall@20} & \textbf{MRR} \\
\midrule
POP (most popular) & 0.12 & 0.02 \\
Item-KNN & 0.26 & 0.08 \\
BPR-MF & 0.31 & 0.10 \\
GRU4Rec & \textbf{0.45} & \textbf{0.16} \\
\bottomrule
\end{tabular}
\end{table}
\end{frame}

%=================================================================
\begin{frame}{Sequential Pattern Mining in Media}
\small
\textbf{Character-level RNN for playlist continuation} (Spotify):

\begin{equation*}
P(\text{track}_{t+1} \mid \text{context}_{<t}) = \text{softmax}(\mathbf{W} \cdot \text{LSTM}(\text{embeddings}_{<t}))
\end{equation*}

\vspace{0.15cm}
\textbf{Types of sequential patterns}:
\begin{enumerate}
    \item \textbf{Artist affinity}: "After listening to Artist A, users often play Artist B"
    \item \textbf{Mood transition}: "High energy $\to$ medium energy $\to$ cool down"
    \item \textbf{Genre flow}: "Hip hop $\to$ R\&B $\to$ Soul" (smooth genre transitions)
    \item \textbf{Session arc}: "Exploration $\to$ peak $\to$ wind-down"
\end{enumerate}

\vspace{0.15cm}
\textbf{Transformer-based session model} (BERT4Rec):
\begin{equation*}
P(i_t \mid i_{<t}, i_{>t}) = \text{softmax}(\mathbf{W} \cdot \text{BERT}(\mathbf{E}_{i_1}, \ldots, \mathbf{E}_{i_T})_t)
\end{equation*}
Bidirectional context: the next recommendation depends on BOTH past and future items in the session.

\vspace{0.15cm}
\textbf{Why it matters}: Playlist listening is sequential by nature --- the next track should fit the current vibe.
\end{frame}

\begin{frame}{Offline Evaluation Metrics for Media}
\small
\begin{columns}[T]

\textbf{Ranking Metrics}
\begin{align*}
\text{Recall@k} &= \frac{|\text{relevant} \cap \text{top-k}|}{|\text{relevant}|} \\

\text{Session Depth} &= \text{avg items/session} \\
\text{Repeat Rate} &= \frac{\#\text{replays}}{\#\text{recommended}}

\section{Fairness \& Diversity}

\begin{frame}{The Filter Bubble Problem}
\small
\textbf{The problem}: Recommendation systems create echo chambers by only recommending content similar to what the user already consumes.

\begin{columns}[T]
\column{0.48\textwidth}
\textbf{Symptoms}:
\begin{itemize}
    \item Niche interests never discovered
    \item Political polarization (YouTube radicalization)
    \item Musical taste stagnation
    \item Reduced catalog coverage
\end{itemize}

\column{0.48\textwidth}
\textbf{Metrics}:
\begin{align*}
\text{Catalog Coverage} &= \frac{|\text{recommended items}|}{|\text{total items}|} \\
\text{Genre Diversity} &= 1 - \text{Gini}(\text{genre distribution}) \\
\text{Serendipity} &= \frac{|\text{rec} \cap \text{novel}|}{|\text{rec}|}
\end{align*}
\end{columns}

\vspace{0.15cm}
\textbf{Diversity-accuracy tradeoff}:
\begin{equation*}
\text{Score}_{\text{final}}(u,i) = \text{Score}_{\text{rel}}(u,i) + \lambda \cdot \underbrace{\text{MMR}(\text{rec}_u, i)}_{\text{diversity penalty}}
\end{equation*}
MMR = Maximum Marginal Relevance: penalizes items too similar to already-recommended ones.
\end{frame}

%=================================================================
\begin{frame}{Debiasing Media Recommendations}
\small
\textbf{Bias sources in media RS}:
\begin{itemize}
    \item \textbf{Popularity bias}: popular items get recommended more $\to$ get even more popular
    \item \textbf{Position bias}: top positions get more clicks regardless of relevance
    \item \textbf{Exposure bias}: we only observe feedback on recommended items
    \item \textbf{Selection bias}: users self-select what they engage with
\end{itemize}

\vspace{0.15cm}
\textbf{Popularity debiasing (Spotify)}:
\begin{equation*}
\text{Score}_{\text{corrected}}(u,i) = \text{Score}_{\text{raw}}(u,i) - \gamma \cdot \log(\text{popularity}_i)
\end{equation*}

\vspace{0.15cm}
\textbf{Inverse Propensity Scoring (IPS)}:
\begin{equation*}
\hat{R}_{\text{IPS}}(\pi) = \frac{1}{|U|}\sum_u\sum_i \frac{\mathbb{I}[\pi(u)=i] \cdot c_{ui}}{P(\text{observed}_{ui})}
\end{equation*}

\vspace{0.15cm}
\textbf{Algorithmic fairness constraints}:
\begin{align*}
\text{Demographic Parity: } & P(\text{rec} \mid A) = P(\text{rec} \mid B) \\
\text{Genre Balance: } & \text{Gini}(\text{rec}_u) \ge \tau
\end{align*}
\end{frame}

\section{The Future}

\begin{frame}{Generative AI \& The Next Frontier}
\small
\textbf{Matching $\to$ Creating}:
\begin{itemize}
    \item \textit{Current}: "Here are songs you might like"
    \item \textit{Next-gen}: "Here's a personalized AI-generated track based on your taste"
\end{itemize}

\vspace{0.15cm}
\textbf{Generative features}:
\begin{enumerate}
    \item \textbf{AI DJ} (Spotify): LLM-powered commentary between tracks
    \item \textbf{Personalized playlists}: "Give me 30min of high-energy workout music"
    \item \textbf{Smart shuffles}: Rearrange queue based on mood transitions
    \item \textbf{Mood detection}: Real-time facial expression $\to$ music matching
\end{enumerate}

\vspace{0.15cm}
\textbf{Architecture}: Retrieval-Augmented Generation
\begin{align*}
\text{Retrieve: } &\mathcal{D}_{\text{rel}} = \text{Top-K}(\text{EBR}(q,\mathcal{I})) \\
\text{Generate: } &\text{Playlist} = \text{LLM}(q \oplus \mathcal{D}_{\text{rel}} \oplus \text{taste\_profile})
\end{align*}

\vspace{0.15cm}
\textbf{Challenges}:
\begin{itemize}
    \item Real-time generation latency
    \item Licensing and copyright
    \item Evaluating generated content quality
\end{itemize}
\end{frame}
